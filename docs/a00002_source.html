500  private string getCurrentTime()
501  {
502  try
503  {
504  DateTime systemTime = DateTime.Now;
505 
506  if (messageFormat.date.dateFormat.Equals(DateFormat.none) &&
507  !messageFormat.time.timeFormat.Equals(TimeFormat.none))
508  {
509  return systemTime.ToString("[" + messageFormat.time.timeFormatString + "] ",
510  CultureInfo.InvariantCulture);
511  }
512  else if (!messageFormat.date.dateFormat.Equals(DateFormat.none) &&
513  messageFormat.time.timeFormat.Equals(TimeFormat.none))
514  {
515  return systemTime.ToString("[" + messageFormat.date.dateFormatString + "] ",
516  CultureInfo.InvariantCulture);
517  }
518  else if (messageFormat.date.dateFormat.Equals(DateFormat.none) &&
519  messageFormat.time.timeFormat.Equals(TimeFormat.none))
520  {
521  return "";
522  }
523  else
524  {
525  return systemTime.ToString("[" + messageFormat.date.dateFormatString + (char)32 +
526  messageFormat.time.timeFormatString + "] ",
527  CultureInfo.InvariantCulture);
528  }
529  }
530  catch(Exception)
531  {
532  return "[INVALID DATE FORMAT] ";
533  }
