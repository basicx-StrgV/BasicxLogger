904  private string getCurrentTime()
905  {
906  try
907  {
908  string start = "";
909  string end = "";
910  if (logFile.type.Equals(LogFileType.txt) || logFile.type.Equals(LogFileType.log))
911  {
912  start = "[";
913  end = "] ";
914  }
915 
916  if (messageFormat.date.dateFormat.Equals(DateFormat.none) &&
917  !messageFormat.time.timeFormat.Equals(TimeFormat.none))
918  {
919  return DateTime.Now.ToString(start + messageFormat.time.timeFormatString + end,
920  messageFormat.time.cultureInfo);
921  }
922  else if (!messageFormat.date.dateFormat.Equals(DateFormat.none) &&
923  messageFormat.time.timeFormat.Equals(TimeFormat.none))
924  {
925  return DateTime.Now.ToString(start + messageFormat.date.dateFormatString + end,
926  messageFormat.time.cultureInfo);
927  }
928  else if (messageFormat.date.dateFormat.Equals(DateFormat.none) &&
929  messageFormat.time.timeFormat.Equals(TimeFormat.none))
930  {
931  return "";
932  }
933  else
934  {
935  return DateTime.Now.ToString(start + messageFormat.date.dateFormatString + (char)32 +
936  messageFormat.time.timeFormatString + end, messageFormat.time.cultureInfo);
937  }
938  }
939  catch (Exception)
940  {
941  return "[INVALID DATETIME FORMAT] ";
942  }
943  }
