703  private string getCurrentTime()
704  {
705  try
706  {
707  string start = "";
708  string end = "";
709  if (logFile.type.Equals(LogFileType.txt) || logFile.type.Equals(LogFileType.log))
710  {
711  start = "[";
712  end = "] ";
713  }
714 
715  if (messageFormat.date.dateFormat.Equals(DateFormat.none) &&
716  !messageFormat.time.timeFormat.Equals(TimeFormat.none))
717  {
718  return DateTime.Now.ToString(start + messageFormat.time.timeFormatString + end,
719  messageFormat.time.cultureInfo);
720  }
721  else if (!messageFormat.date.dateFormat.Equals(DateFormat.none) &&
722  messageFormat.time.timeFormat.Equals(TimeFormat.none))
723  {
724  return DateTime.Now.ToString(start + messageFormat.date.dateFormatString + end,
725  messageFormat.time.cultureInfo);
726  }
727  else if (messageFormat.date.dateFormat.Equals(DateFormat.none) &&
728  messageFormat.time.timeFormat.Equals(TimeFormat.none))
729  {
730  return "";
731  }
732  else
733  {
734  return DateTime.Now.ToString(start + messageFormat.date.dateFormatString + (char)32 +
735  messageFormat.time.timeFormatString + end, messageFormat.time.cultureInfo);
736  }
737  }
738  catch(Exception)
739  {
740  return "[INVALID DATETIME FORMAT] ";
741  }
742  }

859  private void logToXml(Tag messageTag, string message, string id = "")
860  {
861  try
862  {
863  //Load the xml file
864  XmlDocument xmlFile = new XmlDocument();
865  xmlFile.Load(getFullFilePath());
866 
867  //Get the root node from the file
868  XmlNode rootNode = xmlFile.SelectSingleNode(logFile.name);
869 
870  //Create nodes and add data
871  XmlNode logMessageNode = xmlFile.CreateElement("LogMessage");
872  XmlAttribute idAttribute = xmlFile.CreateAttribute("id");
873  if (!id.Equals(""))
874  {
875  idAttribute.Value = id;
876  }
877  logMessageNode.Attributes.Append(idAttribute);
878 
879  XmlNode timestampNode = xmlFile.CreateElement("timestamp");
880  if (!messageFormat.date.dateFormat.Equals(DateFormat.none) ||
881  !messageFormat.time.timeFormat.Equals(TimeFormat.none))
882  {
883  timestampNode.InnerText = getCurrentTime();
884  }
885  logMessageNode.AppendChild(timestampNode);
886 
887  XmlNode tagNode = xmlFile.CreateElement("tag");
888  if (!messageTag.Equals(Tag.none))
889  {
890  tagNode.InnerText = messageTag.ToString();
891  }
892  logMessageNode.AppendChild(tagNode);
893 
894  XmlNode messageNode = xmlFile.CreateElement("message");
895  messageNode.InnerText = message;
896  logMessageNode.AppendChild(messageNode);
897 
898  //Appand to root node
899  rootNode.AppendChild(logMessageNode);
900 
901  //Save the xml file
902  xmlFile.Save(getFullFilePath());
903  }
904  catch (Exception e)
905  {
906  throw e;
907  }

910  private void logToJson(Tag messageTag, string message, string id = "")
911  {
912  try
913  {
914  string fileContent = File.ReadAllText(getFullFilePath());
915 
916  List<LogMessage> logs = JsonSerializer.Deserialize<List<LogMessage>>(fileContent);
917 
918  LogMessage newLog = new LogMessage();
919  if (!id.Equals(""))
920  {
921  newLog.id = id;
922  }
923 
924  if (!messageFormat.date.dateFormat.Equals(DateFormat.none) ||
925  !messageFormat.time.timeFormat.Equals(TimeFormat.none))
926  {
927  newLog.timestamp = getCurrentTime();
928  }
929 
930  if (!messageTag.Equals(Tag.none))
931  {
932  newLog.tag = messageTag.ToString();
933  }
934 
935  newLog.message = message;
936 
937  logs.Add(newLog);
938 
939  string newFileContent = JsonSerializer.Serialize(logs);
940 
941  FileStream fileWriter = File.OpenWrite(getFullFilePath());
942  Utf8JsonWriter jsonWriter = new Utf8JsonWriter(fileWriter, new JsonWriterOptions { Indented = true });
943 
944  JsonDocument jsonFile = JsonDocument.Parse(newFileContent);
945 
946  jsonFile.WriteTo(jsonWriter);
947 
948  jsonWriter.Flush();
949 
950  jsonWriter.Dispose();
951 
952  fileWriter.Close();
953  fileWriter.Dispose();
954  }
955  catch (Exception e)

957  throw e;
958  }
