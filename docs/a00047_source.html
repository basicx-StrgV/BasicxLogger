607  {

609 
610  List<string> idParts = new List<string>();
611 
612  while (idParts.Count != 10)
613  {
614  idParts.Add(new Random().Next(0, 16).ToString("X"));
615  }
616 
617  foreach (string part in idParts)
618  {
619  id = id + part;
620  }
621 
622  return id;
